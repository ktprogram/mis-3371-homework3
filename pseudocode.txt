// Display today's date on page load
FUNCTION displayCurrentDate
    GET element for date display
    FORMAT today's date as "Month Day, Year"
    DISPLAY formatted date in specified element

// Restrict date of birth field to today or earlier
FUNCTION setDOBMaxDate
    GET date input field for DOB
    SET max date to today's date in YYYY-MM-DD format

// Display dynamic slider value
FUNCTION updateSliderValue(value)
    SET slider display element text to current slider value

// Validate form fields on input (on entering or leaving fields)
FUNCTION validateField(field)
    SWITCH field type
        CASE "name fields" (e.g., first, middle, last)
            CHECK for valid characters (letters, dashes, etc.)
            DISPLAY error if invalid
        CASE "date of birth"
            CHECK valid date format and age limit
            DISPLAY error if invalid
        CASE "SSN"
            CHECK for numeric format (9 digits)
            DISPLAY error if invalid
        CASE "phone number"
            CHECK for format ###-###-####
            DISPLAY error if invalid
        CASE "address" fields
            CHECK for non-empty input
            DISPLAY error if invalid
        CASE "email"
            CHECK email format
            DISPLAY error if invalid
        CASE "password"
            CHECK length, special characters, match with confirm password, and no overlap with User ID
            DISPLAY error if invalid
    END SWITCH

// Validate passwords and match confirmation
FUNCTION validatePasswords
    CHECK password meets criteria (length, special chars, etc.)
    CHECK password matches confirm password
    RETURN true if valid, false if not

// Review form data for display
FUNCTION reviewFormData
    GET all form field values
    FORMAT values into readable output
    DISPLAY formatted data in review section

// On form submission, validate all fields and prevent submission if errors exist
EVENT FORM submit
    IF NOT all fields valid THEN
        PREVENT form submission
    END IF

// Initialization on page load
EVENT WINDOW load
    CALL displayCurrentDate
    CALL setDOBMaxDate

// Event listeners for dynamic interactions
ADD event listeners on form fields
    CALL validateField on blur (when leaving field)
ADD event listener to slider
    CALL updateSliderValue on input
